def stage2_subgoal_constraint1(end_effector, keypoints):
    """The box cutter (keypoint 0) should be 10cm above the center of the blue box (average of keypoints 1 and 2)."""
    box_center = 0.5 * (keypoints[1] + keypoints[2])
    offsetted_point = box_center + np.array([0, 0, 0.10])
    return np.linalg.norm(end_effector - offsetted_point)
