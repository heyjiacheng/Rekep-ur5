def stage2_subgoal_constraint1(end_effector, keypoints):
    """The box cutter (keypoint 0) should be 10cm above the center of the blue box (average of keypoints 3 and 4)."""
    box_center = (keypoints[3] + keypoints[4]) / 2
    offsetted_point = box_center + np.array([0, 0, 0.10])
    cost = np.linalg.norm(keypoints[0] - offsetted_point)
